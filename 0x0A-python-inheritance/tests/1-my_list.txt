# 1-my_list.txt

========================
 How to use 1-my_list.py
========================

This modual contain one class Mylist that is derivated from the class list.

MyList Methods
==============
print_sorted(self): Print the list in ascending sorted order


Usage
=====

::
	>>> MyList = __import__('1-my_list').MyList


::
	>>> my_list = MyList()
	>>> my_list.append(1)
	>>> my_list.append(4)
	>>> my_list.append(2)
	>>> my_list.append(3)
	>>> my_list.append(5)
	>>> print(my_list)
	[1, 4, 2, 3, 5]
	>>> my_list.print_sorted()
	[1, 2, 3, 4, 5]
	>>> print(my_list)
	[1, 4, 2, 3, 5]


passed a list
::
	>>> a = [1, 9, 4]
	>>> my_list = MyList(a)
	>>> my_list.print_sorted()
	[1, 4, 9]

empty list
::
	>>> a = []
	>>> my_list = MyList(a)
	>>> my_list.print_sorted()
	[]

unique number
::
	>>> my_list = MyList()
	>>> my_list.append(1)
	>>> my_list.print_sorted()
	[1]
	>>> print(my_list)
	[1]

same number
::
	>>> my_list2 = MyList()
	>>> my_list2.append(1)
	>>> my_list2.append(1)
	>>> my_list2.append(1)
	>>> my_list2.append(1)
	>>> my_list2.print_sorted()
	[1, 1, 1, 1]

negative numbers
::
	>>> MyList = __import__('1-my_list').MyList
	>>> my_list = MyList()
	>>> my_list.append(5)
	>>> my_list.append(1)
	>>> my_list.append(-1)
	>>> my_list.append(51)
	>>> my_list.append(-5)
	>>> my_list.print_sorted()
	[-5, -1, 1, 5, 51]
